# Verify openclaw --version → 2026.3.31

Configure MiniMax M2.7
# Create env file touch ~/.openclaw/.env
MINIMAX_API_HOST=https://api.minimaxi.com MINIMAX_API_KEY=sk-your-key-here TAVILY_API_KEY=tvly-your-key
# Add model to openclaw.json openclaw models add minimax/MiniMax-M2.7
